kitty的一些配置

MacOS上用过iterm2, wezterm, warp,但是用的最习惯的还是kitty

配置基本没怎么动

conf
font_family      Menlo
bold_font        auto
italic_font      auto
bold_italic_font auto

font_size 14.0
window_padding_width 25
include theme.conf

#hide_window_decorations titlebar-only
#background_opacity 0.60
#hide_window_decorations yes
#confirm_os_window_close 0

scrollback_lines  10000
scrollback_indicator_opacity 1.0

map cmd+f combine : show_scrollback : send_text normal,application /

map cmd+shift+h launch --location=hsplit
map cmd+shift+v launch --location=vsplit
map cmd+shift+c close_window

map cmd+left  neighboring_window left  
map cmd+right neighboring_window right  
map cmd+up    neighboring_window up  
map cmd+down  neighboring_window down

map cmd+shift+up move_window up
map cmd+shift+left move_window left
map cmd+shift+right move_window right
map cmd+shift+down move_window down
