#include<iostream>
#include<vector>

using namespace std;

long long int isvalid(int x, int y, vector<vector<long long int>> map) {
    if (map[x][y] == -1) return 0;
    else return map[x][y];
}
int main() {
    int x1, y1, x2, y2;
    cin >> x1 >> y1 >> x2 >> y2;
    vector<vector<long long int>> map(x1 + 1, vector<long long int>(y1 + 1,0));

    map[x2][y2] = -1;
    int dx[] = {1, 2, 2, 1, -1, -2, -2, -1};
    int dy[] = {-2, -1, 1, 2, 2, 1, -1, -2};
    for (int i = 0; i < 8; ++i) {
        int nx = x2 + dx[i];
        int ny = y2 + dy[i];
        if (nx >= 0 && nx <= x1 && ny >= 0 && ny <= y1) {
            map[nx][ny] = -1;
        }
    }

    for (int i = 1; i <= x1; ++i) {
        if(map[i][0]!=-1)map[i][0] = 1;
        else break;
    }
    for (int i = 1; i <= y1; ++i) {
        if(map[0][i]!=-1)map[0][i] = 1;
        else break;
    }

    for (int i = 1; i <= x1; ++i) {
        for (int j = 1; j <= y1; ++j) {
            if (map[i][j] != -1) {
                map[i][j] = isvalid(i - 1, j, map) + isvalid(i, j - 1, map);
            }
        }
    }
    
    cout<<map[x1][y1];
    
}
